    ```python
    class LLMGeneratedDataset(IterableDataset):
        def __init__(self, config, tokenizer, epoch=0, cache_path=None):
            self.config = config
            self.tokenizer = tokenizer
            self.epoch = epoch
            self.cache_path = cache_path  # If set, read from cache instead of LLM

        def __iter__(self):
            if self.cache_path and Path(self.cache_path).exists():
                yield from self._iter_from_cache()
            else:
                yield from self._iter_from_llm()

        def _iter_from_cache(self):
            samples = load_cache(Path(self.cache_path))
            for sample in samples:
                yield sample

        def _iter_from_llm(self):
            worker_info = get_worker_info()
            worker_id = worker_info.id if worker_info else 0
            num_workers = worker_info.num_workers if worker_info else 1
            batch_idx = worker_id
            # Yield up to samples_per_batch samples
            for _ in range(self.config.data.samples_per_batch):
                seed = self.epoch * 10000 + batch_idx * 100 + worker_id
                # Generate sample (sync wrapper around async call)
                sample = self._generate_sample(seed)
                if sample is not None:
                    yield sample
                batch_idx += num_workers
    ```

    The `_generate_sample` method: uses `asyncio.run()` to call the async LLM client (or `asyncio.get_event_loop().run_until_complete()` if already in async context). Calls `build_generation_prompt` with domain from `get_domain_for_seed(seed)`, calls `call_openrouter`, then `parse_ref_tags`, then `char_spans_to_bio`. Returns the encoding dict. On any exception, logs warning and returns None (skip sample).
